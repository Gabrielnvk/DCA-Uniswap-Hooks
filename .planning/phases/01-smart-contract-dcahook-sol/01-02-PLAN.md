    pragma solidity ^0.8.26;

    import {Script, console} from "forge-std/Script.sol";
    import {DCAHook} from "../src/DCAHook.sol";
    import {IPoolManager} from "v4-core/interfaces/IPoolManager.sol";
    import {Hooks} from "v4-core/libraries/Hooks.sol";
    import {HookMiner} from "../test/utils/HookMiner.sol";

    contract DeployDCAHook is Script {
        // Base mainnet PoolManager address
        // IMPORTANT: verify this address on basescan before deploying
        address constant POOL_MANAGER = 0x7Da1D65F8B249183667cdE74C5CBD46dU40159C; // PLACEHOLDER -- replace with actual

        function run() external {
            uint256 deployerPrivateKey = vm.envUint("PRIVATE_KEY");
            address deployer = vm.addr(deployerPrivateKey);

            console.log("Deployer:", deployer);
            console.log("Deployer balance:", deployer.balance);

            // Define required hook flags -- only afterSwap
            uint160 flags = uint160(Hooks.AFTER_SWAP_FLAG);

            // Mine a CREATE2 salt that produces an address with correct flag bits
            console.log("Mining CREATE2 salt for hook address with afterSwap flag...");

            bytes memory creationCode = type(DCAHook).creationCode;
            bytes memory constructorArgs = abi.encode(POOL_MANAGER);

            (address hookAddress, bytes32 salt) = HookMiner.find(
                deployer,
                flags,
                creationCode,
                constructorArgs
            );

            console.log("Found hook address:", hookAddress);
            console.log("Salt:", vm.toString(salt));

            // Deploy using CREATE2
            vm.startBroadcast(deployerPrivateKey);

            DCAHook hook = new DCAHook{salt: salt}(IPoolManager(POOL_MANAGER));

            require(address(hook) == hookAddress, "Hook address mismatch");

            vm.stopBroadcast();

            // Output for frontend integration
            console.log("------------------------------------");
            console.log("DCAHook deployed at:", address(hook));
            console.log("PoolManager:", POOL_MANAGER);
            console.log("Chain: Base (8453)");
            console.log("------------------------------------");

            // Write address to file for frontend consumption
            string memory json = string(abi.encodePacked(
                '{"dcaHook":"', vm.toString(address(hook)),
                '","poolManager":"', vm.toString(POOL_MANAGER),
                '","chainId":8453}'
            ));
            vm.writeFile("deployment.json", json);
            console.log("Deployment info written to deployment.json");
        }
    }
