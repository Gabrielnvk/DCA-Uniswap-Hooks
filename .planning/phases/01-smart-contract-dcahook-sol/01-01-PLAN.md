    ```solidity
    function createDCA(
        PoolKey calldata key,
        bool zeroForOne,
        uint256 amountPerSwap,
        uint256 interval,
        uint256 totalSwaps
    ) external returns (uint256 orderId) {
        require(amountPerSwap > 0, "DCAHook: zero amount");
        require(totalSwaps > 0, "DCAHook: zero swaps");
        require(
            interval == 3600 || interval == 86400 || interval == 604800,
            "DCAHook: invalid interval"
        );

        // Determine tokenIn/tokenOut based on swap direction
        Currency tokenIn = zeroForOne ? key.currency0 : key.currency1;
        Currency tokenOut = zeroForOne ? key.currency1 : key.currency0;

        uint256 totalDeposit = amountPerSwap * totalSwaps;

        // Pull tokens from user (SC-03: upfront deposit, SC-07: SafeERC20, P2)
        IERC20(Currency.unwrap(tokenIn)).safeTransferFrom(
            msg.sender,
            address(this),
            totalDeposit
        );

        orderId = nextOrderId++;

        orders[orderId] = DCAOrder({
            owner: msg.sender,
            tokenIn: tokenIn,
            tokenOut: tokenOut,
            poolKey: key,
            zeroForOne: zeroForOne,
            amountPerSwap: amountPerSwap,
            interval: interval,
            totalSwaps: totalSwaps,
            swapsExecuted: 0,
            lastExecutionTime: block.timestamp,
            deposited: totalDeposit,
            received: 0,
            active: true
        });

        userOrderIds[msg.sender].push(orderId);

        bytes32 poolId = keccak256(abi.encode(key));
        poolOrderIds[poolId].push(orderId);

        emit OrderCreated(
            orderId,
            msg.sender,
            tokenIn,
            tokenOut,
            amountPerSwap,
            interval,
            totalSwaps,
            totalDeposit
        );
    }
    ```

    Key design points:
    - `amountPerSwap` is in the tokenIn's native decimals (SC-09, P1). The frontend must call `IERC20Metadata(token).decimals()` and use `parseUnits(amount, decimals)`. The contract does NOT normalize decimals -- it stores and uses raw amounts as provided.
    - `lastExecutionTime` is set to `block.timestamp` at creation so the first execution happens after one full interval.
    - Interval is restricted to 3600 (hourly), 86400 (daily), or 604800 (weekly) to match requirements.
    - Overflow check: `amountPerSwap * totalSwaps` can overflow for absurdly large values. Solidity 0.8.x has built-in overflow checks so this reverts safely.

    Also add view functions for frontend reads:
    ```solidity
    function getOrder(uint256 orderId) external view returns (DCAOrder memory) {
        return orders[orderId];
    }

    function getUserOrderIds(address user) external view returns (uint256[] memory) {
        return userOrderIds[user];
    }
    ```

    **afterSwap callback** (SC-01, SC-04, SC-08, P3, P11, P12):
    ```solidity
    function afterSwap(
        address,
        PoolKey calldata key,
        IPoolManager.SwapParams calldata,
        BalanceDelta,
        bytes calldata
    ) external override onlyPoolManager returns (bytes4, int128) {
        // P11: Prevent recursive execution -- if we are already executing
        // a DCA swap that triggered another afterSwap on this pool, skip.
        if (_executing) {
            return (this.afterSwap.selector, 0);
        }

        _executing = true;

        bytes32 poolId = keccak256(abi.encode(key));
        uint256[] storage orderIds = poolOrderIds[poolId];
        uint256 executed = 0;
        uint256 maxPerCallback = 5; // P12: gas limit protection

        for (uint256 i = 0; i < orderIds.length && executed < maxPerCallback; i++) {
            DCAOrder storage order = orders[orderIds[i]];

            if (!order.active) continue;
            if (order.swapsExecuted >= order.totalSwaps) continue;
            if (block.timestamp < order.lastExecutionTime + order.interval) continue;

            _executeDCAOrder(orderIds[i], order);
            executed++;
        }

        _executing = false;

        return (this.afterSwap.selector, 0);
    }
    ```

    Note: `onlyPoolManager` modifier comes from `BaseHook` and enforces `msg.sender == address(poolManager)` (SC-08, P3).

    **Internal execution function** (SC-04, P5, P9):
    ```solidity
    function _executeDCAOrder(uint256 orderId, DCAOrder storage order) internal {
        // Checks-effects-interactions (P11):
        // Update state BEFORE external calls
        order.swapsExecuted += 1;
        order.lastExecutionTime = block.timestamp; // P9: resets to now, skips missed intervals

        if (order.swapsExecuted >= order.totalSwaps) {
            order.active = false;
        }

        // Approve PoolManager to take tokenIn for the swap
        IERC20(Currency.unwrap(order.tokenIn)).forceApprove(
            address(poolManager),
            order.amountPerSwap
        );

        // Execute swap through PoolManager
        // The hook holds the tokenIn, swaps it, receives tokenOut
        BalanceDelta delta = poolManager.swap(
            order.poolKey,
            IPoolManager.SwapParams({
                zeroForOne: order.zeroForOne,
                amountSpecified: -int256(order.amountPerSwap), // exact input
                sqrtPriceLimitX96: order.zeroForOne
                    ? TickMath.MIN_SQRT_PRICE + 1
                    : TickMath.MAX_SQRT_PRICE - 1
            }),
            ""  // no hookData for the nested swap
        );

        // P5: Settle deltas -- the hook pays tokenIn and receives tokenOut
        // Settle tokenIn (hook pays what pool expects)
        uint256 amountIn = order.amountPerSwap;

        // Calculate tokenOut received from the delta
        int128 amount0Delta = delta.amount0();
        int128 amount1Delta = delta.amount1();
        uint256 amountOut;
        if (order.zeroForOne) {
            // We sent token0 (negative delta), received token1 (positive delta)
            amountOut = uint256(uint128(amount1Delta));
        } else {
            amountOut = uint256(uint128(amount0Delta));
        }

        // Settle: transfer tokenIn to pool, take tokenOut from pool
        // Use Currency.settle() and Currency.take() helpers
        order.tokenIn.settle(poolManager, address(this), amountIn, false);
        order.tokenOut.take(poolManager, address(this), amountOut, false);

        order.received += amountOut;

        emit OrderExecuted(orderId, amountIn, amountOut, block.timestamp);
    }
    ```

    **IMPORTANT**: The exact delta settlement API depends on the v4-core version in the template. The pattern above uses `Currency.settle()` and `Currency.take()`. If the template's v4-core version uses a different API (e.g., `poolManager.settle()` / `poolManager.take()`), adapt accordingly. Check `lib/v4-core/src/types/Currency.sol` for the available functions.

    Also add this import at the top:
    ```solidity
    import {TickMath} from "v4-core/libraries/TickMath.sol";
    ```

    **Note on sqrtPriceLimitX96**: Using MIN+1 / MAX-1 means no price limit (accepts any price). For a production DCA, you would add a slippage parameter to the order struct (P7). For the demo, this is acceptable.

    Add the `withdrawDCA` function to `DCAHook.sol`:

    ```solidity
    function withdrawDCA(uint256 orderId) external {
        DCAOrder storage order = orders[orderId];

        require(order.owner == msg.sender, "DCAHook: not owner");
        require(order.active, "DCAHook: order not active");

        order.active = false;

        // Calculate remaining tokenIn to refund
        uint256 swapsRemaining = order.totalSwaps - order.swapsExecuted;
        uint256 refundAmount = swapsRemaining * order.amountPerSwap;
        uint256 receivedAmount = order.received;

        // Reset received to 0 since we are withdrawing everything
        order.received = 0;

        // Transfer remaining tokenIn back to owner (SC-07: SafeERC20)
        if (refundAmount > 0) {
            IERC20(Currency.unwrap(order.tokenIn)).safeTransfer(
                order.owner,
                refundAmount
            );
        }

        // Transfer accumulated tokenOut to owner
        if (receivedAmount > 0) {
            IERC20(Currency.unwrap(order.tokenOut)).safeTransfer(
                order.owner,
                receivedAmount
            );
        }

        emit OrderWithdrawn(orderId, refundAmount, receivedAmount);
    }
    ```

    Key design points:
    - Only the order owner can withdraw (access control).
    - Deactivates the order immediately (state change before transfers -- checks-effects-interactions).
    - Returns both remaining unswapped tokenIn AND accumulated tokenOut.
    - Uses `safeTransfer` for both transfers (SC-07, P2).
    - If the order has completed all swaps naturally, user can still call this to collect tokenOut (but `active` will already be `false` from `_executeDCAOrder`, so add a separate claim path).

    Add a separate `claimTokenOut` for completed orders:
    ```solidity
    function claimTokenOut(uint256 orderId) external {
        DCAOrder storage order = orders[orderId];

        require(order.owner == msg.sender, "DCAHook: not owner");
        require(order.received > 0, "DCAHook: nothing to claim");

        uint256 receivedAmount = order.received;
        order.received = 0;

        IERC20(Currency.unwrap(order.tokenOut)).safeTransfer(
            order.owner,
            receivedAmount
        );

        emit OrderWithdrawn(orderId, 0, receivedAmount);
    }
    ```

    This allows claiming output tokens from completed orders where `active` is already `false`.
